import re
from typing import Optional

_PAT_TAG = re.compile(r'@(\w+)')
_PAT_BPM = re.compile(r'\b(\d{2,3})\s*(?:BPM|bpm)\b')
_PAT_KEY = re.compile(r'\b([A-G][#b]?(?:maj|min))\b')


def parse_filename(stem: str) -> dict:
    """Extract tag, bpm, key from stem name. Returns cleaned name with tokens removed."""
    tag_m = _PAT_TAG.search(stem)
    bpm_m = _PAT_BPM.search(stem)
    key_m = _PAT_KEY.search(stem)

    tag = tag_m.group(1) if tag_m else None
    bpm = float(bpm_m.group(1)) if bpm_m else None
    key = key_m.group(1) if key_m else None

    clean = stem
    for pat in (_PAT_TAG, _PAT_BPM, _PAT_KEY):
        clean = pat.sub('', clean)
    clean = re.sub(r'[\s_\-]+', ' ', clean).strip(' _-')

    return {'tag': tag, 'bpm': bpm, 'key': key, 'clean_name': clean}
```

# Camelot Wheel: (note_index, mode) → camelot string
# note_index matches NOTES list: C=0, C#=1, D=2, D#=3, E=4, F=5, F#=6, G=7, G#=8, A=9, A#=10, B=11
_CAMELOT = {
    # Major (B suffix)
    (0,  'maj'): '8B',  (1,  'maj'): '3B',  (2,  'maj'): '10B',
    (3,  'maj'): '5B',  (4,  'maj'): '12B', (5,  'maj'): '7B',
    (6,  'maj'): '2B',  (7,  'maj'): '9B',  (8,  'maj'): '4B',
    (9,  'maj'): '11B', (10, 'maj'): '6B',  (11, 'maj'): '1B',
    # Minor (A suffix)
    (0,  'min'): '5A',  (1,  'min'): '12A', (2,  'min'): '7A',
    (3,  'min'): '2A',  (4,  'min'): '9A',  (5,  'min'): '4A',
    (6,  'min'): '11A', (7,  'min'): '6A',  (8,  'min'): '1A',
    (9,  'min'): '8A',  (10, 'min'): '3A',  (11, 'min'): '10A',
}


def _camelot_key(note: str, mode: str) -> str:
    """Return Camelot Wheel code for a given note and mode ('maj'/'min')."""
    try:
        idx = NOTES.index(note)
    except ValueError:
        return '?'
    return _CAMELOT.get((idx, mode), '?')


def _confidence_from_scores(best: float, all_scores: np.ndarray) -> float:
    """Normalize best Krumhansl score to 0.0–1.0 confidence."""
    min_s = all_scores.min()
    max_s = all_scores.max()
    if max_s == min_s:
        return 0.5
    return float(np.clip((best - min_s) / (max_s - min_s), 0.0, 1.0))
```

def detect_key(y, sr) -> tuple[str, float]:
    """Returns (key_string, confidence) e.g. ('G#min', 0.87)"""
    if sr != 22050:
        y  = librosa.resample(y, orig_sr=sr, target_sr=22050)
        sr = 22050

    chroma      = librosa.feature.chroma_cens(y=y, sr=sr)
    chroma_mean = chroma.mean(axis=1)

    if chroma_mean.sum() > 0:
        chroma_mean = chroma_mean / chroma_mean.sum()

    best_key   = 'C'
    best_mode  = 'maj'
    best_score = -np.inf
    all_scores = []

    for i in range(12):
        maj  = np.roll(_MAJOR, i);  maj  = maj  / maj.sum()
        min_ = np.roll(_MINOR, i);  min_ = min_ / min_.sum()

        s_maj = float(np.dot(chroma_mean, maj))
        s_min = float(np.dot(chroma_mean, min_))
        all_scores.extend([s_maj, s_min])

        if s_maj > best_score:
            best_score, best_key, best_mode = s_maj, NOTES[i], 'maj'
        if s_min > best_score:
            best_score, best_key, best_mode = s_min, NOTES[i], 'min'

    confidence = _confidence_from_scores(best_score, np.array(all_scores))
    return f"{best_key}{best_mode}", confidence
```

def _classify_stem_type(sub_ratio: float, onset_rate: float, centroid: float) -> str:
    """Classify stem type from extracted audio features."""
    if sub_ratio > 0.6:
        return '808'
    if onset_rate > 12 and centroid > 4000:
        return 'hihat'
    if onset_rate > 8 and centroid < 2000:
        return 'kick'
    if 6 <= onset_rate <= 12 and 2000 <= centroid <= 6000:
        return 'snare'
    if onset_rate < 4:
        return 'melodic'
    return 'unknown'


def _build_description(onset_rate: float, centroid: float, bpm: float, key: str) -> str:
    """Build short human-readable beat description."""
    character = 'Percussive' if onset_rate > 6 else 'Tonal'
    brightness = 'bright' if centroid > 3000 else 'dark'
    return f"{character}, {brightness}, {int(bpm)} BPM, {key}"
```

```python
def analyze_file(path: str) -> dict:
    y, sr = librosa.load(path, sr=44100, mono=True)
    bpm   = detect_bpm(path, y, sr)
    key, confidence = detect_key(y, sr)

    # Split key string e.g. 'G#min' → note='G#', mode='min'
    if key.endswith('maj'):
        note, mode = key[:-3], 'maj'
    else:
        note, mode = key[:-3], 'min'

    camelot = _camelot_key(note, mode)

    # Sub energy ratio
    y_low      = librosa.effects.preemphasis(y, coef=-0.97)  # invert = boost lows
    freq_bins  = np.fft.rfftfreq(2048, d=1.0 / sr)
    fft_mag    = np.abs(np.fft.rfft(y, n=2048))
    sub_mask   = freq_bins < 120
    sub_ratio  = float(fft_mag[sub_mask].sum() / (fft_mag.sum() + 1e-8))

    # Onset rate (onsets per second)
    onsets     = librosa.onset.onset_detect(y=y, sr=sr, units='time')
    duration   = librosa.get_duration(y=y, sr=sr)
    onset_rate = len(onsets) / max(duration, 1.0)

    # Spectral centroid (mean Hz)
    centroid   = float(librosa.feature.spectral_centroid(y=y, sr=sr).mean())

    stem_type   = _classify_stem_type(sub_ratio, onset_rate, centroid)
    description = _build_description(onset_rate, centroid, bpm, key)

    return {
        'key':         key,
        'camelot':     camelot,
        'bpm':         bpm,
        'confidence':  round(confidence, 2),
        'stem_type':   stem_type,
        'description': description,
    }
```

import os
from datetime import date
from mutagen.id3 import ID3, TIT2, TPE1, TBPM, TKEY, COMM, TDRC, ID3NoHeaderError
from mutagen.flac import FLAC
from mutagen.oggvorbis import OggVorbis
from mutagen.aiff import AIFF
from mutagen.wave import WAVE


def write_metadata(path: str, data: dict) -> None:
    """Write title, artist, bpm, key, comment, date tags. Silent on error."""
    try:
        ext = os.path.splitext(path)[1].lower()
        if ext == '.mp3':
            _write_id3(path, data)
        elif ext == '.flac':
            _write_flac(path, data)
        elif ext in ('.wav',):
            _write_wave(path, data)
        elif ext in ('.aiff', '.aif'):
            _write_aiff(path, data)
        elif ext == '.ogg':
            _write_ogg(path, data)
        # other formats: silently skip
    except Exception:
        pass  # never crash the rename batch


def _write_id3(path: str, data: dict) -> None:
    try:
        tags = ID3(path)
    except ID3NoHeaderError:
        tags = ID3()
    tags['TIT2'] = TIT2(encoding=3, text=data.get('title', ''))
    tags['TPE1'] = TPE1(encoding=3, text=data.get('artist', ''))
    tags['TBPM'] = TBPM(encoding=3, text=data.get('bpm', ''))
    tags['TKEY'] = TKEY(encoding=3, text=data.get('key', ''))
    tags['COMM'] = COMM(encoding=3, lang='eng', desc='', text=data.get('comment', ''))
    tags['TDRC'] = TDRC(encoding=3, text=data.get('date', ''))
    tags.save(path)


def _write_flac(path: str, data: dict) -> None:
    tags = FLAC(path)
    tags['title']   = data.get('title', '')
    tags['artist']  = data.get('artist', '')
    tags['bpm']     = data.get('bpm', '')
    tags['key']     = data.get('key', '')
    tags['comment'] = data.get('comment', '')
    tags['date']    = data.get('date', '')
    tags.save()


def _write_wave(path: str, data: dict) -> None:
    tags = WAVE(path)
    if tags.tags is None:
        tags.add_tags()
    tags['TIT2'] = TIT2(encoding=3, text=data.get('title', ''))
    tags['TPE1'] = TPE1(encoding=3, text=data.get('artist', ''))
    tags['TBPM'] = TBPM(encoding=3, text=data.get('bpm', ''))
    tags['TKEY'] = TKEY(encoding=3, text=data.get('key', ''))
    tags['COMM'] = COMM(encoding=3, lang='eng', desc='', text=data.get('comment', ''))
    tags['TDRC'] = TDRC(encoding=3, text=data.get('date', ''))
    tags.save()


def _write_aiff(path: str, data: dict) -> None:
    tags = AIFF(path)
    if tags.tags is None:

    tags.tags['TIT2'] = TIT2(encoding=3, text=data.get('title', ''))
    tags.tags['TPE1'] = TPE1(encoding=3, text=data.get('artist', ''))

def _write_ogg(path: str, data: dict) -> None:

```python
import os
import shutil
from typing import Optional


_STEM_FOLDER = {
    'kick':    'Kicks',
    '808':     '808s',
    'hihat':   'Hihats',
    'snare':   'Snares',
    'melodic': 'Melodics',
}


def _bpm_bucket(bpm: Optional[float]) -> str:
    if bpm is None:
        return 'Unsorted'
    if bpm < 100:
        return 'Sub-100'
    if bpm < 120:
        return '100-119 BPM'
    if bpm < 140:
        return '120-139 BPM'
    if bpm < 160:
        return '140-159 BPM'
    return '160plus BPM'


def _stem_folder_name(stem_type: str) -> str:
    return _STEM_FOLDER.get(stem_type, 'Unsorted')


def _source_folder_name(path: str) -> str:
    return os.path.basename(os.path.dirname(os.path.abspath(path))) or 'Unsorted'


def export_files(
    rows: list,
    dest_dir: str,
    structure: str,
    mode: str,
) -> tuple[int, list]:
    """
    Export files to dest_dir organized by structure.
    structure: 'bpm_range' | 'stem_type' | 'source_folder'
    mode:      'copy' | 'move'
    Returns (success_count, error_list).
    """
    success = 0
    errors  = []

    for row in rows:
        path      = row['path']
        bpm       = row.get('bpm')
        stem_type = row.get('stem_type', 'unknown')

        if structure == 'bpm_range':
            subfolder = _bpm_bucket(bpm)
        elif structure == 'stem_type':
            subfolder = _stem_folder_name(stem_type)
        else:  # source_folder
            subfolder = _source_folder_name(path)

        target_dir = os.path.join(dest_dir, subfolder)
        os.makedirs(target_dir, exist_ok=True)
        target_path = os.path.join(target_dir, os.path.basename(path))

        # avoid overwrite
        if os.path.exists(target_path):
            base, ext = os.path.splitext(os.path.basename(path))
            counter = 1
            while os.path.exists(os.path.join(target_dir, f"{base}_{counter}{ext}")):
                counter += 1
            target_path = os.path.join(target_dir, f"{base}_{counter}{ext}")

        try:
            if mode == 'copy':
                shutil.copy2(path, target_path)
            else:
                shutil.move(path, target_path)
            success += 1
        except Exception as e:
            errors.append(f"{os.path.basename(path)}: {e}")

    return success, errors
```

def _export(self):
    dlg = ExportDialog(self)
    if dlg.exec_() != QDialog.Accepted:
        return
    vals = dlg.get_values()
    if not vals['dest']:
        self.status.error("No destination folder selected.")
        return

    rows = []
    for row in range(self.table.rowCount()):
        path = self._file_paths.get(row)
        if not path:
            continue
        bpm_t = self.table.item(row, COL_BPM).text()
        try:
            bpm = float(bpm_t)
        except ValueError:
            bpm = None
        stem_info = getattr(self, '_stem_data', {}).get(row, {})
        rows.append({
            'path':      path,
            'bpm':       bpm,
            'stem_type': stem_info.get('stem_type', 'unknown'),
            'key':       self.table.item(row, COL_KEY).text(),
        })

    self.status.pulse()
    success, errors = export_files(rows, vals['dest'], vals['structure'], vals['mode'])

    if errors:
        self.status.error(f"{success} exported, {len(errors)} failed.")
        QMessageBox.warning(self, "Export errors", "\n".join(errors))
    else:
        self.status.done(f"{success} file(s) exported to {vals['dest']}")
